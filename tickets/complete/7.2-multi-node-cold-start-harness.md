## Usage

```ts

const mesh = await createMesh(3, {
  responsibilityK: 3,
  clusterSize: 3,
  superMajorityThreshold: 0.67
});
const transactors = buildNetworkTransactors(mesh);

const transactorA = transactors.get(mesh.nodes[0].peerId.toString())!;
const treeA = await Tree.createOrOpen(transactorA, 'my-tree', e => e.key);
await treeA.replace([[1, { key: 1, value: 'a' }]]);
```
